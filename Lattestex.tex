% syntax highlighting
% base code style with listings
\lstdefinestyle{codebase}{
    %backgroundcolor=\color{black},
    basicstyle=\ttfamily\footnotesize,
    keywordstyle=\color{blue085085255},
    stringstyle=\color{greenGovIE000077068},
    commentstyle=\color{yellow}\ttfamily,
    morecomment=[l][\color{orange}]{\#},
    showspaces=false,
    showstringspaces=false,
    showtabs=false,
    tabsize=2,
    %numbers=left, % line numbers
    numberstyle=\footnotesize, % line numbers font size
    stepnumber=1, % line numbers step between two line numbers
    numbersep=10pt, % line numbers distance from code
    escapeinside={\%*}{*)}, % comment within code
    breaklines=true,
    breakatwhitespace=false,
    %frame=single,
    frame=tb,
    %caption="Code",
    captionpos=b,
}
% Python listings style
\lstdefinestyle{pythonstyle}{
    style=codebase,
    language=Python,
    emph={MyClass,__init__},
    emphstyle=\color{red255085085},
}
% LaTeX listings style
\lstdefinestyle{latexstyle}{
    style=codebase,
    language=TeX,
    emph={},
    emphstyle=\color{red255085085},
}
% C++ listings style
\lstdefinestyle{cppstyle}{
    style=codebase,
    language=C++,

}
